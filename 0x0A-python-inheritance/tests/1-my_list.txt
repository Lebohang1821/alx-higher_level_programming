# Import the MyList class from the 1-my_list module
MyList = __import__('1-my_list').MyList

# Common case: Create a MyList instance, append values, print and print_sorted
my_list = MyList()
my_list.append(1)
my_list.append(4)
my_list.append(2)
my_list.append(3)
my_list.append(5)
print(my_list)  # Expect [1, 4, 2, 3, 5]
my_list.print_sorted()  # Expect [1, 2, 3, 4, 5]
print(my_list)  # Expect [1, 4, 2, 3, 5]

# Try to access an undefined object (my_list2)
MyList = __import__('1-my_list').MyList
try:
    my_list2.print_sorted()
except NameError as e:
    print(e)  # Expect "name 'my_list2' is not defined"

# Create a MyList instance with a list, print_sorted
a = [1, 9, 4]
my_list = MyList(a)
my_list.print_sorted()  # Expect [1, 4, 9]

# Create a MyList instance with an empty list, print_sorted
a = []
my_list = MyList(a)
my_list.print_sorted()  # Expect []

# Append a unique number, print and print_sorted
my_list = MyList()
my_list.append(1)
my_list.print_sorted()  # Expect [1]
print(my_list)  # Expect [1]

# Append the same number multiple times, print_sorted
my_list2 = MyList()
my_list2.append(1)
my_list2.append(1)
my_list2.append(1)
my_list2.append(1)
my_list2.print_sorted()  # Expect [1, 1, 1, 1]

# Try to call print_sorted on a list object (not MyList)
my_list = [1, 2, 3]
try:
    my_list.print_sorted()
except AttributeError as e:
    print(e)  # Expect "'list' object has no attribute 'print_sorted'"

# Append negative numbers, print_sorted
MyList = __import__('1-my_list').MyList
my_list = MyList()
my_list.append(5)
my_list.append(1)
my_list.append(-1)
my_list.append(51)
my_list.append(-5)
my_list.print_sorted()  # Expect [-5, -1, 1, 5, 51]
